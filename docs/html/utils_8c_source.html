46/**
47 * @brief Unified factorial dispatcher
48 */
49bool factorial(double_t n, double_t* result, uint8_t flags) {
50 // Common validation
51 if (n < 0 || isnan(n)) return false;
52
53 // Forced method cases
54 if (flags & FLAG_FORCE_LOOKUP) {
55 int int_n = (int)n;
56 if (int_n != n) return false;
57 uint64_t exact;
58 if (!lut_factorial(int_n, &exact)) return false;
59 *result = (double_t)exact;
60 return true;
61 }
62 if (flags & FLAG_FORCE_GAMMA) return gamma_factorial(n, result);
63 if (flags & FLAG_FORCE_STIRLING) return stirlings_factorial(n, result);
64
65 // Automatic selection
66 if ((flags & FLAG_PREFER_EXACT) || n == (int)n) {
67 int int_n = (int)n;
68
69 // Exact integer cases
70 if (int_n <= 20) {
71 uint64_t exact;
72 if (lut_factorial(int_n, &exact)) {
73 *result = (double_t)exact;
74 return true;
75 }
76 }
77
78 // Larger but still exact in double_t
79 if (int_n <= 170) {
80 return gamma_factorial(n, result);
81 }
82
83 // Too large for exact representation
84 if (flags & FLAG_PREFER_EXACT) return false;
85 }
86
87 // Fallback to Stirling's
88 return stirlings_factorial(n, result);
